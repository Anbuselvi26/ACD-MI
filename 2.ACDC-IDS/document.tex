	\section{Motivation \& Background}
	
	\begin{frame}{Motivation: The Interpretability Gap in Network Security}
		\begin{itemize}
			\setstretch{1.6}
			\item \justifying Modern network security systems increasingly rely on
			\textbf{deep learning } models for automated threat detection.
			
			\item \justifying \textbf{The Black-Box Problem:} These models generate alerts,
			but their \textbf{internal decision logic } is opaque to human analysts.
			
			\item \justifying \textbf{Operational Impact:} Security teams face
			\textbf{high false-positive rates}.
			
			\item \justifying \textbf{Key Challenge:} Understanding \emph{why} a model
			raises an alert, not just \emph{that} it raises one.
			
			\item \justifying Motivation:
			\textbf{causal interpretability} > correlational explanations. \tiny \cite{ahmed2021explainable,xin2022xaiids}
			
		\end{itemize}
	\end{frame}

	\begin{frame}{Applying ACDC to NIDS: Core Idea}
		\setstretch{1.6}
		\begin{itemize}
			\item In NIDS, the target behavior is an \textbf{attack detection decision}
			(e.g., SYN flood, scan). 
			
			\item \justifying ACDC treats the trained NIDS as a \textbf{signal-processing system}
			mapping traffic statistics to decisions.
			
			\item\justifying The goal is to identify the \textbf{minimal internal computation}.
			
			\item\justifying The extracted circuit reveals the \textbf{features and interactions}
			that causally drive detection. \tiny\cite{conmy2023acdc,lyle2023efficient}
			
		\end{itemize}
		
	\end{frame}
	
	% --- SECTION 2: TECHNICAL FRAMEWORK ---
	\section{Technical Framework}
	
	\begin{frame}{High-Level System Architecture}
		\begin{itemize}
			\setstretch{1.6}
			\item \textbf{Data Plane:} Real-time packet ingestion and feature engineering.
			\item \textbf{Inference Plane:} Modular neural Network Intrusion Detection System.
			\item \textbf{Diagnostic Plane:} On-demand ACDC engine for causal analysis.
			\item \justifying Planes are \textbf{decoupled} to preserve throughput and reliability.
		\end{itemize}
		\vspace{0.3cm}
		\centering
		\textit{Interpretability is treated as a diagnostic service, not an inline requirement.} \tiny \cite{ahmed2021explainable}
		
	\end{frame}
	
	\begin{frame}{Data Plane: Ingestion and Feature Engineering}
		\begin{itemize}
			\setstretch{1.4}
			\item \textbf{Packet Broker:} Zeek / Suricata tap live network traffic.
			\item \textbf{Stream Processor:} Kafka or Spark Streaming aggregates packets into flows.
			\item \textbf{Computed Features:}
			\begin{itemize}
				\item Packet rate
				\item Protocol flags
				\item Byte and payload statistics
			\end{itemize}
			\item \textbf{State Store:} Redis or Cassandra maintains temporal context
			(e.g., SYN count over last 10 seconds).
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Inference Plane: Modular Neural NIDS}
		\begin{itemize}
			\setstretch{1.5}
			\item Model implemented as a \textbf{modular computational graph}.
			\item Each logical feature group corresponds to a \textbf{distinct sub-network}.
			\item Enables node-level intervention required by ACDC.
			\item \textbf{Shadow Mode:}
			\begin{itemize}
				\item Model runs continuously.
				\item High-risk alerts trigger logging of internal activations.
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Diagnostic Plane: ACDC Engine}
		\begin{itemize}
			\setstretch{1.3}
			\item Activated \textbf{only for high-confidence alerts}.
			\item Receives:
			\begin{itemize}
				\item Flow identifier
				\item Full internal activation trace
			\end{itemize}
			\item \textbf{Activation Patching Controller:}
			\begin{itemize}
				\item Injects baseline (benign) activations.
				\item Performs counterfactual tests.
			\end{itemize}
			\item \textbf{Circuit Extractor:}
			\begin{itemize}
				\item Prunes non-causal edges.
				\item Outputs minimal causal subgraph.
			\end{itemize}
		\end{itemize}
	\end{frame}

	\begin{frame}{Automated Circuit Discovery Loop}
		\begin{enumerate}
			\setstretch{1.3}
			\item Input: Malicious flow $x_m$, benign flow $x_b$.
			\item Forward pass: record all node activations.
			\item Patch each edge with benign activations.
			\item Measure output divergence (KL divergence).
			\item If divergence $< \epsilon$, mark edge as non-causal.
			\item Recursively prune graph.
		\end{enumerate}
		\vspace{0.2cm}
		\centering
		\textit{Result: a minimal circuit sufficient to explain the alert.}
	\end{frame}

	% Slide 10
	\begin{frame}{Step B: Pruning Criterion ($\tau$)}
		\justifying ACDC removes an internal connection ($w \rightarrow v$) if it does not
		causally influence the attack detection outcome.
		
		\vfill
		\begin{equation*}
			D_{KL}(G \parallel H_{\text{new}}) - D_{KL}(G \parallel H) < \tau
		\end{equation*}
		\vfill
		
		\begin{itemize}
			\item \textbf{$G$}: Full NIDS model predicting attack vs. normal traffic.
			\item \textbf{$H$}: Current candidate circuit under evaluation.
			\item \textbf{$D_{KL}$}: Change in output probability distribution.
		\end{itemize}
		
		\vspace{0.4cm}
		\justifying KL divergence ensures the circuit preserves the \textbf{decision logic},
		not merely classification accuracy.
	\end{frame}
	
	\begin{frame}{Step C: Activation Patching}
		\begin{itemize}
			\setstretch{1.4}
			\item \justifying Internal activations from \textbf{corrupted (benign) traffic}
			are injected into the forward pass of \textbf{malicious traffic}.
			
			\item \justifying If the attack prediction remains unchanged, the patched
			connection is \textbf{not causally required} for detection.
			
			\item \justifying This allows ACDC to distinguish
			\textbf{true attack logic} from incidental correlations.
			
			\item \justifying \textbf{Example:} replacing SYN packet-rate activations
			with statistics from normal background traffic.
		\end{itemize}
	\end{frame}

	\section{Research Applications}
	
	% Slide 13
	\begin{frame}{Research: Generalization vs. Spurious Correlation}
		\begin{itemize}
			\setstretch{1.6}
			\item Does the trained NIDS learn \textbf{attack-relevant traffic patterns}
			(e.g., TCP flags, packet-rate bursts, temporal structure)?
			
			\item Or does it rely on \textbf{incidental identifiers}
			(e.g., source IPs, ports, or dataset-specific artifacts)?
			
			\item ACDC reveals whether the extracted circuit encodes
			\textbf{robust attack semantics} or shortcut-based correlations. \cite{ahmed2021explainable}
		
		\end{itemize}
	\end{frame}
	
	% Slide 14
	\begin{frame}{Example: Circuit Discovery for DDoS Detection}
		\begin{itemize}
			\setstretch{1.6}
			\item \justifying Consider a trained NIDS tasked with detecting \textbf{volumetric DDoS attacks}.
			
			\item ACDC isolates a \textbf{minimal detection circuit} that:
			\begin{itemize}
				\item aggregates packet-rate signals across short time windows,
				\item emphasizes sudden temporal traffic bursts,
				\item suppresses protocol-specific and benign background noise.
			\end{itemize}
			
			\item \justifying This circuit constitutes the \textbf{causal computation}
			sufficient for triggering a DDoS alert.
		\end{itemize}
	\end{frame}

	% Slide 18
	\begin{frame}{Logic Redundancy Discovery: The OR-Gate Problem}
		\begin{itemize}
			\setstretch{1.5}
			
			\item \justifying \textbf{Redundancy:}
			A NIDS may learn multiple internal pathways that independently detect the same attack.
			
			\item \justifying \textbf{Pruning Risk:}
			If removing one pathway does not affect performance, ACDC may prune both during causal testing.
			
			\item \justifying \textbf{Interpretation:}
			This reflects redundancy in detection logic—not model failure.
			
			\item \textbf{Research Challenge:}
			Handling redundant causal pathways remains an open problem in circuit discovery.
		\end{itemize}
	\end{frame}
	
	% Slide 17
	\begin{frame}{Scalability via an Asynchronous Forensic Pipeline}
		\begin{itemize}
			\setstretch{1.5}
			\item \justifying \textbf{Decoupled Design:}
			ACDC functions as a diagnostic service, separate from inline detection.
			
			\item \justifying \textbf{Parallel Execution:}
			Real-time inference and causal circuit discovery run in independent planes.
			
			\item \justifying \textbf{No Impact on Inline Detection:}
			Activation patching is isolated, preserving throughput and latency.
			
			\item \justifying \textbf{Efficient Discovery:}
			Attribution patching reduces computational cost.
			
			\item \textbf{Flow:}
			\textit{Detect} $\rightarrow$ \textit{Trigger} $\rightarrow$ \textit{Explain}.
		\end{itemize}
	\end{frame}

	\begin{frame}{Circuit Level Debugging for NIDS Models}
		\setstretch{1.3}
		\begin{itemize}
			\item \textbf{The Bottleneck:} Traditional NIDS are "Black Boxes"—fixing one error often requires retraining the entire model.
			
			\vspace{0.6em}
			
			\item \textbf{ACDC-Enabled Debugging:}
			\begin{itemize}
				\item \textbf{Pinpoint:} Identify the specific sub-circuit responsible for a false alert.
				\item \textbf{Causal Validation:} Confirm the error using activation patching with "clean" baseline traffic.
				\item \textbf{Targeted Update:} Refine only the implicated nodes, leaving the rest of the model's logic intact.
			\end{itemize}
			
			\vspace{0.6em}
			
			\item \textbf{Operational Value:} 
			\begin{itemize}
				\item Faster time-to-patch for SOC teams.
				\item Eliminates the risk of "catastrophic forgetting" during global retraining.
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Evaluation: Quantifying Model Trust}
		\setstretch{1.4}
		\begin{itemize}
			\item \textbf{How do we know the refined circuit is reliable?}
			\vspace{0.8em}
			\item \textbf{Faithfulness ($D_{KL}$):} Does the simplified circuit still "think" like the full model?
			\item \textbf{Robustness (Accuracy):} Does detection remain high after we prune the "noise"?
			\item \textbf{Interpretability (Sparsity):} Is the circuit small enough for a human to audit in under 60 seconds? \tiny \cite{conmy2023acdc}
			
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Operational Validation in NIDS}
		\setstretch{1.4}
		\begin{itemize}
			\item \textbf{Recovery Quality:} Proof that the circuit captures the \textit{actual} attack signatures, not just random noise.
			
			\vspace{0.5em}
			
			\item \textbf{Corrupted Traffic Stress Test:} 
			\begin{itemize}
				\item Does the circuit break if we corrupt a malicious feature?
				\item If yes, the circuit has successfully identified a \textbf{causal feature}.
			\end{itemize}
			
			\vspace{0.5em}
			
			\item \textbf{Q1/Q2 Logical Integrity:}
			\begin{itemize}
				\item \textbf{Q1 (Sufficiency):} Is the circuit enough to detect the attack?
				\item \textbf{Q2 (Minimality):} Does removing any more edges break the detection?
			\end{itemize}
		\end{itemize}
	\end{frame}
