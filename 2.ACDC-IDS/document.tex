	%------------------------------------------------
	% Motivation: ACDC for NIDS
	%------------------------------------------------
	\begin{frame}{Motivation: Explainability in NIDS}
		\begin{itemize}
			\setstretch{1.4}
			\item Deep learning based NIDS achieve high detection accuracy
			\item However, modern NIDS models behave as \textbf{black boxes}
			\item Existing explainability methods (SHAP, LIME):
			\begin{itemize}
				\item provide feature attribution
				\item lack causal guarantees
				\item fail to explain \emph{internal computation}
			\end{itemize}
			\item Security-critical systems require:
			\begin{itemize}
				\item trustworthy decisions
				\item auditability
				\item forensic interpretability
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	%------------------------------------------------
	% Behavior Definition
	%------------------------------------------------
	\begin{frame}{Behavior Definition in NIDS}
		\begin{itemize}
			\setstretch{1.4}
			\item In mechanistic interpretability, a \textbf{behavior} is:
			\begin{itemize}
				\item a well-defined input–output pattern
				\item measurable via a task-specific metric
			\end{itemize}
			\item In NIDS, behaviors correspond to:
			\begin{itemize}
				\item DDoS detection
				\item Port scanning detection
				\item Brute-force or probe attacks
			\end{itemize}
			\item Each behavior is isolated using:
			\begin{itemize}
				\item attack-specific datasets
				\item prediction confidence or logit difference
			\end{itemize}
		\end{itemize}
	\end{frame}

	%------------------------------------------------
	% Circuit Concept
	%------------------------------------------------
	\begin{frame}{What Is a Circuit?}
		\begin{itemize}
			\setstretch{1.4}
			\item A \textbf{circuit} is a sparse subgraph of the model
			\item It captures the internal computation responsible for a behavior
			\item In transformers: attention heads and MLP blocks
			\item In NIDS models:
			\begin{itemize}
				\item feature interaction paths
				\item temporal dependencies
				\item hidden neuron activations
			\end{itemize}
			\item Goal: isolate the minimal computation needed for attack detection
		\end{itemize}
	\end{frame}
	
	%------------------------------------------------
	% Scalability Challenge
	%------------------------------------------------
	\begin{frame}{Scalability Challenges in Circuit Discovery}
		\begin{itemize}
			\setstretch{1.4}
			\item Traditional circuit discovery relies on:
			\begin{itemize}
				\item manual inspection
				\item human intuition
				\item trial-and-error patching
			\end{itemize}
			\item This approach does not scale to:
			\begin{itemize}
				\item deep NIDS architectures
				\item multiple attack classes
				\item large feature spaces
			\end{itemize}
			\item \textbf{ACDC automates this step}
		\end{itemize}
	\end{frame}
	
	%------------------------------------------------
	% ACDC Overview
	%------------------------------------------------
	\begin{frame}{Automatic Circuit Discovery (ACDC)}
		\begin{itemize}
			\setstretch{1.4}
			\item ACDC is a post-hoc, causal interpretability method
			\item Operates on a \textbf{trained and frozen model}
			\item Key idea:
			\begin{itemize}
				\item represent the model as a computational graph
				\item iteratively remove causally irrelevant connections
			\end{itemize}
			\item Output: minimal, behavior-preserving circuit
		\end{itemize}
	\end{frame}
	
	%------------------------------------------------
	% ACDC Pipeline for NIDS
	%------------------------------------------------
	\begin{frame}{ACDC Pipeline for NIDS}
		\begin{itemize}
			\setstretch{1.4}
			\item Step 1: Define attack-specific behavior
			\item Step 2: Construct computational graph of NIDS
			\item Step 3: Generate clean and corrupted traffic
			\item Step 4: Apply activation patching
			\item Step 5: Recursively prune unimportant edges
			\item Step 6: Extract attack-specific circuit
		\end{itemize}
	\end{frame}
	
	%------------------------------------------------
	% Activation Patching
	%------------------------------------------------
	\begin{frame}{Activation Patching}
		\begin{itemize}
			\setstretch{1.4}
			\item Activation patching tests causal importance
			\item Replace internal activations with corrupted versions
			\item Measure impact on attack prediction
			\item If prediction unchanged:
			\begin{itemize}
				\item component is not causally necessary
			\end{itemize}
			\item Forms the basis of edge removal in ACDC
		\end{itemize}
	\end{frame}
