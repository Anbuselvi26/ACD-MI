	\section{Motivation \& Background}
	
	% Slide 2
	\begin{frame}{The Interpretability Gap in NIDS}
		\begin{itemize}
			\setstretch{1.3}
			\item \justifying Modern NIDS use Deep Learning (Transformers/RNNs) for high accuracy.
			\item \justifying \textbf{The Black-Box Problem:} Analysts receive alerts but cannot verify the causal logic.
			\item \justifying \textbf{SOC Impact:} High false-positive rates and lack of trust in "AI-driven" security.
			\item \justifying \textbf{Solution:} Mechanistic Interpretability to reverse-engineer model internals.
			\item \justifying This motivates causal—not correlational—explanations.
		\end{itemize}
	\end{frame}

	% Slide 4
	\begin{frame}{Applying ACDC to NIDS: Core Idea}
		\begin{itemize}
			\setstretch{1.3}
			\item ACDC is applied \textbf{post-hoc} to a trained NIDS model
			\item \justifying The NIDS model is treated as a computational system, not a black box
			\item The key objective is to identify:
			\begin{itemize}
				\item which internal components
				\item and which information pathways
			\end{itemize}
			are \textbf{causally responsible} for detecting a specific attack
			\item \justifying This is achieved by systematically testing and pruning the model’s
			internal computational graph
		\end{itemize}
	\end{frame}

	% Slide 6
	\begin{frame}{The NIDS Computational Graph}
		\begin{itemize}
			\setstretch{1.4}
			\item Represent the NIDS model as a Directed Acyclic Graph (DAG).
			\item \textbf{Nodes ($V$):} Residual stream, Attention Heads, MLP layers.
			\item \textbf{Edges ($E$):} Linear information flow between components (e.g., residual connections, attention outputs).
			
			\item \textbf{Task:} Prune $E$ until only the "Security Logic" remains.
		\end{itemize}
		\vspace{0.3cm}
		This abstraction is architecture-agnostic (Transformer, RNN, CNN).
		
	\end{frame}

	% Slide 9
	\begin{frame}{Step A: Task and Dataset Setup}
		To find a circuit, we must define the behavior:
		\vspace{0.5cm}
		\begin{block}{Task Definition}
			Detecting SYN Flood attacks within the UNSW-NB15 dataset.
		\end{block}
		\vspace{0.5cm}
		\begin{itemize}
			\item \textbf{Clean Dataset:} Malicious traffic samples.
			\item \textbf{Corrupted Dataset:} Normal baseline traffic or mean-ablated features.
		\end{itemize}
		\vspace{0.25cm}
		\justifying Behavior is quantified using the logit difference between Attack and Normal classes.
	\end{frame}
	
	% Slide 10
	\begin{frame}{Step B: The Pruning Criterion ($\tau$)}
		ACDC removes an edge ($w \to v$) if the change in performance is below a threshold $\tau$.
		\vfill
		\begin{equation*}
			D_{KL}(G \parallel H_{new}) - D_{KL}(G \parallel H) < \tau
		\end{equation*}
		\vfill
		\begin{itemize}
			\item \textbf{$G$}: Full NIDS Model.
			\item \textbf{$H$}: Current Subgraph.
			\item \textbf{$D_{KL}$}: Kullback–Leibler Divergence.
		\end{itemize}
		\vspace{0.5cm}
		KL divergence captures distributional change, not just confidence drop.
	\end{frame}
	
	% Slide 11
	\begin{frame}{Step C: Activation Patching}
		\begin{itemize}
			\setstretch{1.5}
			\item Patch activations from corrupted traffic into clean traffic flow.
			\item If prediction is unchanged, the patched edge is \textbf{not} causally required.
			\item This establishes causality rather than correlation.
			\item \justifying \textbf{Example:} replacing SYN packet-rate activations with benign statistics.
		\end{itemize}
	\end{frame}
	
	% Slide 12
	\begin{frame}{The ACDC Algorithm}
		\begin{algorithmic}[1]
			\setstretch{1.2}
			
			\State \textbf{Input:} Model $G$, threshold $\tau$, task $T$
			\State \textbf{Initialize:} $H \gets G$
			\For{each edge $e \in H$ (reverse topological order)}
			\If{metric impact $< \tau$}
			\State $H \gets H \setminus \{e\}$
			\EndIf
			\EndFor
			\State \Return $H$ (minimal security circuit)
		\end{algorithmic}
	\end{frame}
	
	% --- SECTION 3: RESEARCH APPLICATIONS ---
	\section{Research Applications}
	
	% Slide 13
	\begin{frame}{Research: Generalization vs. Spurious Correlation}
		\begin{itemize}
			\setstretch{1.5}
			\item Does the model recognize \textbf{TCP flags} (generalization)?
			\item Or is it relying on specific \textbf{source identifiers} (spurious correlation)?
			\item ACDC exposes whether detection logic is robust or shortcut-based.
			\item This limitation exposes redundancy rather than model failure.
		\end{itemize}
	\end{frame}
	
	% Slide 14
	\begin{frame}{Example: Circuit Discovery for DDoS Detection}
		\begin{itemize}
			\setstretch{1.6}
			\item Consider a trained NIDS detecting volumetric DDoS attacks.
			\item ACDC isolates a circuit that:
			\begin{itemize}
				\item aggregates packet-rate features,
				\item emphasizes short-term temporal bursts,
				\item suppresses protocol-specific noise.
			\end{itemize}
			\item The circuit explains \emph{how} the model detects flooding behavior.
		\end{itemize}
	\end{frame}

	% Slide 16
	\begin{frame}{Research: Adversarial Attack Resilience}
		\begin{itemize}
			\setstretch{1.6}
			\item \justifying Adversarial traffic often induces \textbf{pathway divergence}, activating circuits normally used for benign traffic.
			\item \justifying Research Question: To what extent can adversarial perturbations reroute computation away from the attack circuit while normal-traffic pathways remain active?
			\item ACDC enables circuit-level robustness analysis.
		\end{itemize}
	\end{frame}
	
	% Slide 17
	\begin{frame}{Technical Challenge: Scalability}
		\begin{itemize}
			\setstretch{1.5}
			\item \justifying Activation patching is computationally expensive for high-throughput traffic.
			\item \justifying \textbf{Research Focus:} Attribution patching to approximate circuits efficiently.
			%\item \textbf{Research Focus:} Attribution patching to approximate circuits in a single backward pass.
			\item ACDC is intended for offline analysis, not inline detection.
		\end{itemize}
	\end{frame}
	
	% Slide 18
	\begin{frame}{Failure Case: The OR-Gate Problem}
		\begin{itemize}
			\item Redundant detection pathways may exist for the same attack.
			\item ACDC may prune both if individual removals do not reduce performance.
			\item This remains an open research challenge.
		\end{itemize}
	\end{frame}
