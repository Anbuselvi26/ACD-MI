	\section{Motivation \& Background}
	
	\begin{frame}{Motivation: The Interpretability Gap in Network Security}
		\begin{itemize}
			\setstretch{1.4}
			\item \justifying Modern network security systems increasingly rely on
			\textbf{deep learning } models for automated threat detection.
			
			\item \justifying \textbf{The Black-Box Problem:} These models generate alerts,
			but their \textbf{internal decision logic } is opaque to human analysts.
			
			\item \justifying \textbf{Operational Impact:} Security teams face
			\textbf{high false-positive rates}.
			
			\item \justifying \textbf{Key Challenge:} Understanding \emph{why} a model
			raises an alert, not just \emph{that} it raises one.
			
			\item \justifying Motivation:
			\textbf{causal interpretability} > correlational explanations.
		\end{itemize}
	\end{frame}
	
	% Slide 2
	\begin{frame}{What is a Network Intrusion Detection System (NIDS)}
		\justifying
		\setstretch{1.3}
		\begin{itemize}
			\item \justifying A \textbf{Network Intrusion Detection System (NIDS)} monitors
			network traffic to identify \textbf{malicious activities or intrusions}.
			
			\item \justifying It analyzes \textbf{packets, flows, and traffic patterns}
			traversing the network.
			
			\item \justifying NIDS raises alerts when observed traffic deviates from
			\textbf{normal or expected behavior}.
			
			\item \justifying It supports \textbf{Security Operations Centers (SOC)}
			in detecting attacks such as DoS, scanning, malware, and data exfiltration.
		\end{itemize}
		
		\vspace{0.3cm}
		\textbf{Key idea:}
		\begin{center}
			``Is this network traffic normal or malicious?''
		\end{center}
	\end{frame}

	% Slide 4
	\begin{frame}{Applying ACDC to NIDS: Core Idea}
		\begin{itemize}
			\setstretch{1.4}
			\item In NIDS, the target behavior is an \textbf{attack detection decision}
			(e.g., SYN flood, scan, exfiltration).
			
			\item ACDC treats the trained NIDS as a \textbf{signal-processing system}
			that transforms raw traffic statistics into security decisions.
			
			\item The objective is to identify the \textbf{minimal internal computation}
			that maps:
			\begin{itemize}
				\item traffic patterns (rates, flags, temporal bursts)
				\item to an attack-specific alert.
			\end{itemize}
			
			\item The extracted circuit reveals \textbf{which traffic characteristics}
			and \textbf{which internal interactions} the NIDS truly relies on.
		\end{itemize}
	\end{frame}

	% Slide 6
	\begin{frame}{The NIDS Computational Graph}
		\begin{itemize}
			\setstretch{1.4}
			\item \justifying The trained NIDS is modeled as a \textbf{directed computational graph}.
			\vspace{0.3cm}
			\item \justifying \textbf{Nodes ($V$):} internal processing units such as
			feature interaction layers, temporal aggregation modules,
			attention heads, or hidden neurons.
			
			\item \justifying\textbf{Edges ($E$):} linear information pathways that propagate
			traffic signals (e.g., packet rates, flags, burst patterns)
			across the network.
			
			\item \justifying \textbf{Objective:} identify and retain only those edges that are	\textbf{causally necessary} for a specific attack detection behavior.
		\end{itemize}
		
		\vspace{0.4cm}
		\justifying This abstraction enables circuit discovery across different NIDS architectures
		(Transformer, RNN, CNN).
	\end{frame}

	% Slide 10
	\begin{frame}{Step B: Pruning Criterion ($\tau$)}
		\justifying ACDC removes an internal connection ($w \rightarrow v$) if it does not
		causally influence the attack detection outcome.
		
		\vfill
		\begin{equation*}
			D_{KL}(G \parallel H_{\text{new}}) - D_{KL}(G \parallel H) < \tau
		\end{equation*}
		\vfill
		
		\begin{itemize}
			\item \textbf{$G$}: Full NIDS model predicting attack vs. normal traffic.
			\item \textbf{$H$}: Current candidate circuit under evaluation.
			\item \textbf{$D_{KL}$}: Change in output probability distribution.
		\end{itemize}
		
		\vspace{0.4cm}
		\justifying KL divergence ensures the circuit preserves the \textbf{decision logic},
		not merely classification accuracy.
	\end{frame}
	
	\begin{frame}{Step C: Activation Patching}
		\begin{itemize}
			\setstretch{1.4}
			\item \justifying Internal activations from \textbf{corrupted (benign) traffic}
			are injected into the forward pass of \textbf{malicious traffic}.
			
			\item \justifying If the attack prediction remains unchanged, the patched
			connection is \textbf{not causally required} for detection.
			
			\item \justifying This allows ACDC to distinguish
			\textbf{true attack logic} from incidental correlations.
			
			\item \justifying \textbf{Example:} replacing SYN packet-rate activations
			with statistics from normal background traffic.
		\end{itemize}
	\end{frame}
	
	% --- SECTION 3: RESEARCH APPLICATIONS ---
	\section{Research Applications}
	
	% Slide 13
	\begin{frame}{Research: Generalization vs. Spurious Correlation}
		\begin{itemize}
			\setstretch{1.5}
			\item Does the train NIDS learn \textbf{attack-relevant traffic patterns}
			(e.g., TCP flags, packet-rate bursts, temporal structure)?
			
			\item Or does it rely on \textbf{incidental identifiers}
			(e.g., source IPs, ports, or dataset-specific artifacts)?
			
			\item ACDC reveals whether the extracted circuit encodes
			\textbf{robust attack semantics} or shortcut-based correlations.
			
			\item This analysis helps distinguish \textbf{true generalization}
			from spurious detection behavior.
		\end{itemize}
	\end{frame}
	
	% Slide 14
	\begin{frame}{Example: Circuit Discovery for DDoS Detection}
		\begin{itemize}
			\setstretch{1.6}
			\item \justifying Consider a trained NIDS tasked with detecting \textbf{volumetric DDoS attacks}.
			
			\item ACDC isolates a \textbf{minimal detection circuit} that:
			\begin{itemize}
				\item aggregates packet-rate signals across short time windows,
				\item emphasizes sudden temporal traffic bursts,
				\item suppresses protocol-specific and benign background noise.
			\end{itemize}
			
			\item \justifying This circuit constitutes the \textbf{causal computation}
			sufficient for triggering a DDoS alert.
		\end{itemize}
	\end{frame}

	% Slide 17
	\begin{frame}{Technical Challenge: Scalability}
		\setstretch{1.5}
		\justifying
		
		\begin{itemize}
			\item \justifying  Activation patching is computationally expensive
			for high-throughput network traffic.
			
			\item \justifying \textbf{Research Focus:}
			Attribution patching to approximate circuit importance
			with significantly reduced computational cost.
			
			\item \justifying As a result, ACDC is currently suited for
			\textbf{offline analysis and post-incident investigation},
			rather than inline detection.
		\end{itemize}
	\end{frame}
	
	% Slide 18
	\begin{frame}{Failure Case: The OR-Gate Problem}
		\begin{itemize}
			\setstretch{1.4}
			\item \justifying A NIDS may contain \textbf{redundant internal pathways}
			that independently support the same attack detection.
			
			\item \justifying If removing one pathway does not degrade detection performance,
			ACDC may incorrectly prune both during causal testing.
			
			\item \justifying This does not indicate model failure, but highlights
			\textbf{redundancy in the learned detection logic}.
			
			\item \justifying Addressing such redundancy remains an open research problem
			in circuit discovery methods.
			\item \justifying This is a limitation of greedy causal pruning, not a model error.
		\end{itemize}
	\end{frame}

	\begin{frame}{Automated Forensics for Tier-3 Analysts}
		\begin{itemize}
			\setstretch{1.3}
			\item \justifying For each detected attack, ACDC extracts the
			\textbf{internal detection circuit} used by the NIDS.
			
			\item \justifying Analysts can inspect the \textbf{behavioral logic}
			(e.g., traffic patterns and internal interactions)
			that triggered the alert.
			
			\item \justifying This supports \textbf{faster root-cause analysis}
			for advanced and previously unseen threats.
		\end{itemize}
	\end{frame}
	
	% Slide 22
	\begin{frame}{Logic-Based Filter Tuning and Evaluation}
		\begin{itemize}
			\setstretch{1.5}
			\item Circuit inspection identifies causes of false positives.
			\item Model refinement is applied only to implicated sub-components.
			\item Evaluation metrics:
			\begin{itemize}
				\setstretch{1.2}
				\item KL divergence under ablation,
				\item detection accuracy preservation,
				\item circuit sparsity.
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Evaluating ACDC-Based NIDS: Faithfulness}
		\begin{itemize}
			\setstretch{1.4}
			
			\item \textbf{Faithfulness:}
			Does the extracted circuit behave like the full NIDS?
			
			\item \justifying \textbf{KL Divergence:}
			Low divergence indicates preservation of detection logic.
			
			\item \justifying \textbf{Logit Margin:}
			Drop in attack–benign confidence reveals causally important edges.
			
			\item \textbf{Sparsity:}
			Fewer retained edges imply higher interpretability.
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Grounded Evaluation in NIDS}
		\begin{itemize}
			\setstretch{1.4}
			
			\item \textbf{Recovery Quality:}
			Does the circuit retain true attack-relevant logic?
			
			\item \justifying \textbf{ROC-AUC:}
			Measures separation between relevant and irrelevant components.
			
			\item \justifying \textbf{Clean vs Corrupted Traffic:}
			Correct circuits detect attacks only when malicious features are present.
			
			\item \textbf{Q1 / Q2 Check:}
			Correct logic retained, spurious logic excluded.
		\end{itemize}
	\end{frame}
	
	% Slide 23
	\begin{frame}{Building Operational Trust}
